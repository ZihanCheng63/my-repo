# RuyiSDK 链表实例代码

## 一、核心代码实现
```c
#include <stdio.h>
#include <stdlib.h>

// 定义链表节点结构
typedef struct ListNode {
    int data;
    struct ListNode *next;
} ListNode;

// 创建新节点
ListNode* create_node(int data) {
    ListNode *new_node = (ListNode*)malloc(sizeof(ListNode));
    if (new_node == NULL) {
        perror("malloc failed to create node");
        return NULL;
    }
    new_node->data = data;
    new_node->next = NULL;
    return new_node;
}

// 尾部插入节点
int insert_node_tail(ListNode **head, int data) {
    ListNode *new_node = create_node(data);
    if (new_node == NULL) {
        return -1;
    }
    if (*head == NULL) {
        *head = new_node;
        return 0;
    }
    ListNode *temp = *head;
    while (temp->next != NULL) {
        temp = temp->next;
    }
    temp->next = new_node;
    return 0;
}

// 遍历打印链表
void traverse_list(ListNode *head) {
    if (head == NULL) {
        printf("List is empty!\n");
        return;
    }
    printf("List elements: ");
    ListNode *temp = head;
    while (temp != NULL) {
        printf("%d ", temp->data);
        temp = temp->next;
    }
    printf("\n");
}

// 释放链表内存
void free_list(ListNode **head) {
    if (*head == NULL) {
        return;
    }
    ListNode *temp = NULL;
    while (*head != NULL) {
        temp = *head;
        *head = (*head)->next;
        free(temp);
    }
    *head = NULL;
    printf("List memory freed successfully!\n");
}

// 主函数：演示链表完整操作流程
int main() {
    ListNode *head = NULL;
    int ret = 0;

    ret = insert_node_tail(&head, 10);
    ret += insert_node_tail(&head, 20);
    ret += insert_node_tail(&head, 30);
    ret += insert_node_tail(&head, 40);
    ret += insert_node_tail(&head, 50);

    if (ret != 0) {
        printf("Some nodes insert failed!\n");
        free_list(&head);
        return -1;
    }

    traverse_list(head);

    insert_node_tail(&head, 60);
    printf("After insert 60: ");
    traverse_list(head);

    free_list(&head);
    traverse_list(head);

    printf("RISC-V Link List Demo Run Successfully!\n");
    return 0;
}
```

## 二、编译与运行步骤
### 交叉编译
`riscv64-plct-linux-gnu-gcc -o link_list_demo link_list_demo.c -Wall -O2 -march=rv64imac -mabi=lp64`

### QEMU模拟运行
`qemu-riscv64 -L $(riscv64-plct-linux-gnu-gcc -print-sysroot) ./link_list_demo`
